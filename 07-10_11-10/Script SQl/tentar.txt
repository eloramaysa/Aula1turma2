select *from Vendas where Year(vendas.DatInc) = 2020 order by Vendas.Quantidade desc 

select  YEAR(ver.DatInc), MAX(ver.Quantidade) as quantidade,  mar.Nome  from Marcas mar 
inner join Carros car on car.Marca = mar.Id AND MAR.ID = ( SELECT top 1 MAR2.ID from Marcas mar2 
inner join Carros car2 on car2.Marca = mar2.Id 
inner join Vendas ver2 on VER2.Carro = car2.Id  
order by ver2.Quantidade desc  )
inner join Vendas ver on ver.Carro = car.Id 
group by  Year(ver.DatInc), mar.Nome order by mar.Nome desc


select year(ver.DatInc) , max(ver.Quantidade) as Quantidade, mar.Nome from Vendas ver inner join Carros car on
ver.Carro = car.Id inner join Marcas mar on car.Marca = mar.Id AND MAR.ID = ( SELECT top 1 MAR2.ID from Marcas mar2 
inner join Carros car2 on car2.Marca = mar2.Id 
inner join Vendas ver2 on VER2.Carro = car2.Id  
where Year(ver.DatInc) = max(year(ver2.DatInc))
order by ver2.Quantidade*ver2.Valor desc )
group by  Year(ver.DatInc), mar.Nome order by mar.Nome desc
